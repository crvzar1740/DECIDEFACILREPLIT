    <script>
        lucide.createIcons();
        document.getElementById('current-year').innerText = new Date().getFullYear();

        function openCheckout() {
            document.getElementById('checkout-modal').classList.remove('hidden');
            document.body.style.overflow = 'hidden';
        }

        function closeCheckout() {
            document.getElementById('checkout-modal').classList.add('hidden');
            document.body.style.overflow = '';
        }

        let activeFaqIndex = null;
        function toggleFaq(index) {
            const currentAnswer = document.getElementById(`faq-answer-${index}`);
            const currentIcon = document.getElementById(`faq-icon-${index}`);

            if (activeFaqIndex === index) {
                currentAnswer.classList.add('hidden');
                currentIcon.style.transform = 'rotate(0deg)';
                activeFaqIndex = null;
            } else {
                if (activeFaqIndex !== null) {
                    document.getElementById(`faq-answer-${activeFaqIndex}`).classList.add('hidden');
                    document.getElementById(`faq-icon-${activeFaqIndex}`).style.transform = 'rotate(0deg)';
                }
                currentAnswer.classList.remove('hidden');
                currentIcon.style.transform = 'rotate(180deg)';
                activeFaqIndex = index;
            }
        }
